Sample scenario for testing.

import importlib.util
import os
import xml.etree.ElementTree

import qppsim.AppProfile
import qppsim.Application
import qppsim.Des
import qppsim.Time
import qppsim.Ue


def _try_import(module_path, class_):
    """
    Attempt to import `class_` from the module
    found at the path specified by `module_path`

    If `class_` is not found at the module found
    at `module_path`, then this method will search
    up the path attempting to find `class_`

    :param module_path:
        The absolute Python path
        to the module to search for `class_`.
    :type module_path: str

    :param class_:
        The name of the class to attempt to import
    :type class_: str

    :return:
        The class instance of `class_` if found
        None otherwise
    :rtype: type or None
    """
    path_components = module_path.split('.')

    # Walk our way down the provided path
    # (i.e. from "qppsim.accesscontrol.AccessControlBase" to "qppsim.accesscontrol")
    # Attempting to import the specified class
    for path_end in range(len(path_components), 0, -1):
        import_path = ".".join(path_components[:path_end])
        module_spec = importlib.util.find_spec(import_path)
        if module_spec is not None:
            module = importlib.import_module(module_spec.name)
            found_class = getattr(module, class_)
            return found_class
    return None



[docs]def parse_scenario(filename):
    """
    Parse an XML file with an scenario description. The expected items are:
        - DES
        - UEs
        - Application Profiles
        - Application Instances
    Returns a tuple with the DES, the UEs in a dictionary (key is UE name),
    the application profiles in a dictionary (key is the profile name), and
    the application instances in a dictionary (key is the instance name)
    """
    des = None
    application_profiles = {}
    ues = {}
    application_instances = {}
    imsis = []

    try:
        tree = xml.etree.ElementTree.parse(filename)
    except (EnvironmentError, xml.parsers.expat.ExpatError) as err:
        print("{0}: import error: {1}".format(
            os.path.basename(filename), err))
        raise EnvironmentError(err)

    # DES
    des_entry = tree.find("des")
    if des_entry is not None:
        num_rbs_element = des_entry.find('num_rbs')
        if num_rbs_element is None:
            num_rbs = 50
        else:
            num_rbs = int(num_rbs_element.text)

        start_packet_id_element = des_entry.find('start_packet_id')
        if start_packet_id_element is None:
            start_packet_id = 0
        else:
            start_packet_id = int(start_packet_id_element.text)

        default_qci_element = des_entry.find('default_qci')
        if default_qci_element is None:
            default_qci = 9
        else:
            default_qci = int(default_qci_element.text)
            assert 0 < default_qci < 10, "Invalid QCI value ({0})!".format(
                default_qci)

        default_arp_element = des_entry.find('default_arp')
        if default_arp_element is None:
            default_arp = 11
        else:
            default_arp = int(default_arp_element.text)
            assert 0 < default_arp < 16, "Invalid ARP value ({0})!".format(
                default_arp)

        default_mbr_element = des_entry.find('default_mbr')
        if default_mbr_element is None:
            default_mbr = 1024*1024*1024*1024
        else:
            default_mbr = int(default_mbr_element.text)

        rtx_threshold_element = des_entry.find('rtx_threshold')
        if rtx_threshold_element is None:
            rtx_threshold = 0.1
        else:
            rtx_threshold = float(rtx_threshold_element.text)
            assert 0 <= rtx_threshold <= 1, "Invalid RTX Threshold ({0})!".format(
                rtx_threshold)

        bearer_stats_window_size_element = des_entry.find('bearer_stats_window_size')
        if bearer_stats_window_size_element is None:
            bearer_stats_window_size = qppsim.Time.Time(seconds=60)
        else:
            bearer_stats_window_size = int(bearer_stats_window_size_element.text)

        seed_element = des_entry.find('seed')
        if seed_element is None:
            seed = 60
        else:
            seed = int(seed_element.text)

        stop_time_element = des_entry.find('stop_time')
        if stop_time_element is None:
            stop_time = qppsim.Time.Time(seconds=1)
        else:
            stop_time = qppsim.Time.Time(seconds=float(stop_time_element.text))
            assert stop_time > qppsim.Time.ZERO_TIME, "Invalid Stop Time ({0})!".format(stop_time.milliseconds)

        priority_policy_element = des_entry.find('priority_policy')
        if priority_policy_element is None:
            priority_policy = "PriorityPolicySample"
        else:
            priority_policy = priority_policy_element.text.strip()

        priority_policy_class = _try_import("qppsim.prioritypolicy." + priority_policy, priority_policy)
        if priority_policy_class is None:
            raise ImportError("Failed to import {0}".format(priority_policy))
        priority_policy_instance = priority_policy_class()

        access_control_policy_element = des_entry.find('access_control_policy')
        if access_control_policy_element is None:
            access_control_policy = "AccessControlTraceOnly"
        else:
            access_control_policy = access_control_policy_element.text.strip()

        access_control_policy_class = _try_import("qppsim.accesscontrol." + access_control_policy, access_control_policy)
        if access_control_policy_class is None:
            raise ImportError("Failed to import {0}".format(access_control_policy))
        access_control_policy_instance = access_control_policy_class()

        preemption_policy_element = des_entry.find('preemption_policy')
        if preemption_policy_element is None:
            preemption_policy = "PreemptionDummy"
        else:
            preemption_policy = preemption_policy_element.text.strip()
        preemption_policy_class = _try_import("qppsim.preemption." + preemption_policy, preemption_policy)
        if preemption_policy_class is None:
            raise ImportError("Failed to import {0}".format(preemption_policy))
        preemption_policy_instance = priority_policy_class()

        qos_monitor_element = des_entry.find('qos_monitor')
        if qos_monitor_element is None:
            qos_monitor = "QosMonitorDummy"
        else:
            qos_monitor = qos_monitor_element.text.strip()
        qos_monitor_class = _try_import("qppsim.qosmonitor." + qos_monitor, qos_monitor)
        if qos_monitor_class is None:
            raise ImportError("Failed to import {0}".format(qos_monitor))
        qos_monitor_instance = qos_monitor_class()

        scheduler_element = des_entry.find('scheduler')
        if scheduler_element is None:
            scheduler = "SchedulerRoundRobin"
        else:
            scheduler = scheduler_element.text.strip()
        scheduler_class = _try_import("qppsim.scheduler." + scheduler, scheduler)
        if scheduler_class is None:
            raise ImportError("Failed to import {0}".format(scheduler))
        scheduler_instance = scheduler_class(num_rbs=num_rbs)

        output_dir_element = des_entry.find('output_dir')
        if output_dir_element is None:
            output_dir = "."
        else:
            output_dir = output_dir_element.text.strip()

        topology_filename_element = des_entry.find('topology_filename')
        if topology_filename_element is None:
            topology_filename = "topology.txt"
        else:
            topology_filename = topology_filename_element.text.strip()

        app_traffic_filename_element = des_entry.find('app_traffic_filename')
        if app_traffic_filename_element is None:
            app_traffic_filename = "appTraffic.txt"
        else:
            app_traffic_filename = app_traffic_filename_element.text.strip()

        bearer_filename_element = des_entry.find('bearer_filename')
        if bearer_filename_element is None:
            bearer_filename = "bearerTrace.txt"
        else:
            bearer_filename = bearer_filename_element.text.strip()

        arp_filename_element = des_entry.find('arp_filename')
        if arp_filename_element is None:
            arp_filename = "arpTrace.txt"
        else:
            arp_filename = arp_filename_element.text.strip()

        qos_filename_element = des_entry.find('qos_filename')
        if qos_filename_element is None:
            qos_filename = "qosTrace.txt"
        else:
            qos_filename = qos_filename_element.text.strip()

        trace_qos_element = des_entry.find('trace_qos')
        if trace_qos_element is None:
            trace_qos = False
        else:
            trace_qos = trace_qos_element.text.strip().lower() == "true"

        preempt_qos_element = des_entry.find('preempt_qos')
        if preempt_qos_element is None:
            preempt_qos = False
        else:
            preempt_qos = preempt_qos_element.text.strip().lower() == "true"

        qos_monitor_interval_element = des_entry.find('qos_monitor_interval')
        if qos_monitor_interval_element is None:
            qos_monitor_interval = qppsim.Time.ONE_SECOND
        else:
            qos_monitor_interval = float(qos_monitor_interval_element.text)
            assert qos_monitor_interval > qppsim.Time.ZERO_TIME, "Invalid QoS Monitor Interval ({0})!".format(
                qos_monitor_interval)

        des = qppsim.Des.Des(num_rbs, start_packet_id, default_qci,
                             default_arp, default_mbr, rtx_threshold,
                             bearer_stats_window_size, seed, stop_time,
                             priority_policy_instance, access_control_policy_instance,
                             preemption_policy_instance, qos_monitor_instance, scheduler_instance,
                             output_dir, topology_filename, app_traffic_filename, bearer_filename,
                             arp_filename, qos_filename, trace_qos, preempt_qos,
                             qos_monitor_interval)

    # UEs
    for ue_entry in tree.findall("ue"):
        imsi = int(ue_entry.find('imsi').text)
        assert imsi > 0, "Invalid IMSI ({0})!".format(imsi)
        assert imsi not in imsis, "IMSI {0} already used!".format(imsi)
        imsis.append(imsi)

        name = ue_entry.find("name").text
        assert name not in ues, "UE name '{0}' already used!".format(name)

        mcs = int(ue_entry.find('mcs').text)
        assert 0 <= mcs <= 28, "Invalid MCS ({0})!".format(mcs)

        queue_size_element = ue_entry.find('queue_size')
        if queue_size_element is None:
            queue_size = 10000
        else:
            queue_size = int(queue_size_element.text)
            assert queue_size > 0, "Invalid RLC queue size ({0})!".format(queue_size)

        ues[name] = qppsim.Ue.Ue(imsi, name, mcs, queue_size)

    # App Profiles
    for app_profile_entry in tree.findall("app_profile"):
        name = app_profile_entry.find("name").text
        assert name not in application_profiles, "AppProfile name '{0}' already used!".format(name)

        packet_size_entry = app_profile_entry.find("packet_size")
        packet_size_distribution = packet_size_entry.find("distribution").text.strip()
        packet_size_parameters = []
        for parameter_entry in packet_size_entry.findall("parameter"):
            packet_size_parameters.append(float(parameter_entry.text))

        packet_interval_entry = app_profile_entry.find("packet_interval")
        packet_interval_distribution = packet_interval_entry.find("distribution").text.strip()
        packet_interval_parameters = []
        for parameter_entry in packet_interval_entry.findall("parameter"):
            packet_interval_parameters.append(float(parameter_entry.text))

        packets_session_entry = app_profile_entry.find("packets_session")
        packets_session_distribution = packets_session_entry.find("distribution").text.strip()
        packets_session_parameters = []
        for parameter_entry in packets_session_entry.findall("parameter"):
            packets_session_parameters.append(float(parameter_entry.text))

        session_interval_entry = app_profile_entry.find("session_interval")
        session_interval_distribution = session_interval_entry.find("distribution").text.strip()
        session_interval_parameters = []
        for parameter_entry in session_interval_entry.findall("parameter"):
            session_interval_parameters.append(float(parameter_entry.text))

        application_profiles[name] = qppsim.AppProfile.AppProfile(
            name, [packet_size_distribution, packet_size_parameters],
            [packet_interval_distribution, packet_interval_parameters],
            [packets_session_distribution, packets_session_parameters],
            [session_interval_distribution, session_interval_parameters]
            )

    # Application Instances
    for app_instance_entry in tree.findall("app_instance"):
        name = app_instance_entry.find("name").text
        assert name not in application_instances, "App Instance name '{0}' already used!".format(name)

        profile_name = app_instance_entry.find('profile_name').text.strip()
        assert profile_name in application_profiles, "Profile '{0}' for Instance '{1}' does not exist!".format(
            profile_name, name)

        start_time = qppsim.Time.Time(seconds=float(app_instance_entry.find('start_time').text))
        assert start_time >= qppsim.Time.ZERO_TIME, "Invalid Start Time ({0})!".format(start_time)

        stop_time = qppsim.Time.Time(seconds=float(app_instance_entry.find('stop_time').text))
        assert stop_time >= start_time, "Invalid Stop Time ({0})!".format(stop_time)

        ue_name = app_instance_entry.find('ue_name').text.strip()
        assert ue_name in ues, "UE '{0}' for App Instance '{1}' does not exist!".format(
            ue_name, name)

        default_bearer_element = app_instance_entry.find('default_bearer')
        if default_bearer_element is None:
            default_bearer = False
        else:
            default_bearer = default_bearer_element.text.strip().lower() == "true"

        application_instances[name] = application_profiles[profile_name].create_app(
            name, start_time, stop_time, ues[ue_name], default_bearer)

    return des, application_profiles, ues, application_instances
